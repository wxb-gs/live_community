MySQL 表：notes

CREATE TABLE notes (
    id            BIGINT PRIMARY KEY,          -- Leaf 号段
    user_id       BIGINT NOT NULL,
    title         VARCHAR(200) NOT NULL,
    cover_url     VARCHAR(500),               -- 封面图 MinIO URL
    body_md_key   VARCHAR(500),               -- md 文件 MinIO objectKey
    status        TINYINT DEFAULT 0,          -- 0:draft 1:published 2:deleted
    tags          VARCHAR(500),               -- JSON数组 ["tag1","tag2"]
    location      VARCHAR(200),               -- 地理位置
    like_count    INT DEFAULT 0,
    comment_count INT DEFAULT 0,
    view_count    INT DEFAULT 0,
    share_count   INT DEFAULT 0,
    collect_count INT DEFAULT 0,
    created_at    DATETIME DEFAULT CURRENT_TIMESTAMP,
    updated_at    DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,

    INDEX idx_user_id (user_id),
    INDEX idx_status_created (status, created_at DESC),
    INDEX idx_tags (tags)
);

Cassandra 表：comments

CREATE TABLE comments (
    note_id     BIGINT,          -- partition key
    comment_id  BIGINT,          -- clustering key (Leaf 雪花, 时间有序)
    user_id     BIGINT,
    parent_id   BIGINT,          -- 0=顶级评论, 非0=回复某评论
    reply_to    BIGINT,          -- 回复的目标用户ID
    content     TEXT,
    like_count  INT DEFAULT 0,
    created_at  TIMESTAMP,
    PRIMARY KEY (note_id, comment_id)
) WITH CLUSTERING ORDER BY (comment_id DESC);

-- 查询：SELECT * FROM comments WHERE note_id=? LIMIT 20;
-- 单篇笔记所有评论在同一分区，按时间倒序

getNoteDetail 并行查询

public NoteDetailResponse getNoteDetail(Long noteId) {
    CompletableFuture<Note> noteFuture =
        CompletableFuture.supplyAsync(() -> noteRepo.findById(noteId));  // MySQL

    CompletableFuture<List<Comment>> commentsFuture =
        CompletableFuture.supplyAsync(() -> commentRepo.findByNoteId(noteId, 20));  // Cassandra

    Note note = noteFuture.join();
    List<Comment> comments = commentsFuture.join();

    return NoteDetailResponse.of(note, comments);
}
// MySQL 和 Cassandra 并行查，减少总延迟
